Test de diagnóstico

Abre la consola (F12) y pega estos comandos uno por uno:

// 1. Verificar si jQuery está disponible
console.log('jQuery version:', typeof jQuery !== 'undefined' ? jQuery.fn.jquery : 'NO DISPONIBLE');

// 2. Buscar el script cargado
var scripts = document.querySelectorAll('script[src*="fix-patient-urls"]');
console.log('Scripts encontrados:', scripts.length);
scripts.forEach(function(s) { console.log('  - ' + s.src); });

// 3. Verificar si el CSS está inyectado
var injectedCSS = document.getElementById('cabinetmedfix-responsive-css');
console.log('CSS inyectado:', injectedCSS ? 'SÍ' : 'NO');

// 4. Contar columnas en la tabla
var table = document.querySelector('table.liste');
if (table) {
    var cols = table.querySelectorAll('thead th').length;
    console.log('Columnas en la tabla:', cols);
} else {
    console.log('No se encontró tabla.liste');
}

// 5. Verificar si la clase está aplicada
var wrapper = document.querySelector('.cabinetmed-patients-list');
console.log('Wrapper con clase aplicada:', wrapper ? 'SÍ' : 'NO');
